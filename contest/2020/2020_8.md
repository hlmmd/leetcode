```cpp
class Solution
{
public:
    int countGoodTriplets(vector<int> &arr, int a, int b, int c)
    {
        int count = 0;
        for (int i = 0; i < arr.size() - 2; i++)
        {
            for (int j = i + 1; j < arr.size() - 1; j++)
            {
                if (fabs(arr[i] - arr[j]) > a)
                    continue;
                for (int k = j + 1; k < arr.size(); k++)
                {
                    if (fabs(arr[i] - arr[k]) > c || fabs(arr[j] - arr[k]) > b)
                        continue;
                    count++;
                }
            }
        }

        return count;
    }
};
```

class Solution
{
public:
    int minSwaps(vector<vector<int>> &grid)
    {
        int n = grid.size();
        vector<int> rowZeroCount(n, 0);
        for (int i = 0; i < n; i++)
        {
            for (int j = n - 1; j >= 1; j--)
            {
                if (grid[i][j] == 0)
                    rowZeroCount[i]++;
                else
                    break;
            }
        }
        int rtn = 0;
        for (int k = n - 1; k >= 1; k--)
        {
            int s = n - k - 1;
            int i = s;
            for (; i < n; i++)
            {
                if (rowZeroCount[i] >= k)
                {
                    rtn += i - s;
                    for (int x = i; x > s; x--)
                    {
                        rowZeroCount[x] = rowZeroCount[x - 1];
                    }
                    break;
                }
            }
            if (i == n)
                return -1;
        }

        return rtn;
    }
};

class Solution
{
public:
    string makeGood(string s, int sz = 0)
    {
        while (sz != s.size())
        {
            sz = s.size();
            for (int i = 0; i + 1 < s.size(); ++i)
                if (s[i] != s[i + 1] && tolower(s[i]) == tolower(s[i + 1]))

        }

### [1545. Find Kth Bit in Nth Binary String](https://leetcode.com/problems/find-kth-bit-in-nth-binary-string/)

class Solution
{
public:
    char findKthBit(int n, int k)
    {
        if (n == 1)
            return '0';

        int len = (1 << n) - 1;

        if (k == len / 2 + 1)
        {
            return '1';
        }
        else if (k > len / 2 + 1)
        {
            return findKthBit(n - 1, len - k + 1) == '1' ? '0' : '1';
        }
        else
            return findKthBit(n - 1, k);
    }
};

class Solution
{
public:
    bool threeConsecutiveOdds(vector<int> &arr)
    {
        if (arr.size() < 3)
            return false;
        for (int i = 0; i < arr.size() - 2; i++)
        {
            if ((arr[i] % 2 == 1) && (arr[i + 1] % 2 == 1) && (arr[i + 2] % 2 == 1))
                return true;
        }
        return false;
    }
};

class Solution
{
public:

    int minDays(int n)
    {
        if (n <= 1)
            return n;
        if (dp.count(n) == 0)

class Solution
{
public:
    vector<int> mostVisited(int n, vector<int> &rounds)
    {
        int start = rounds[0];
        int end = rounds.back();
        vector<int> ret;
        if (end >= start)
        {
            for (int i = start; i <= end; i++)
                ret.push_back(i);
        }
        else if (end < start)
        {
            for (int i = 1; i <= end; i++)
                ret.push_back(i);
            for (int i = start; i <= n; i++)
                ret.push_back(i);
        }

        return ret;
    }
};
